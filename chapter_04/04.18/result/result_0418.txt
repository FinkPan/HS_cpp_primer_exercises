//编写程序，使用指针把一个 int 型数组的所有元素设置为 0。 

#include <iostream>
using namespace std;

int main()
{
	int a[] = {1,2,3,4,5,6,7};
	int *p1 = a;
	for (int i =0; i != 6; ++i,++p1)
		*p1 = 0;
	for (int i =0; i != 6; ++i)
		cout << "a[" << i << "]; " << a[i] << endl;
	system("pause");
	return 0;
}
